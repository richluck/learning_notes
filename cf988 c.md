```
permutation随机排列满足连续两项相加为合数：
首先，奇偶性相同的两个数相加是合数，那么只需要找到最小的连续的一个奇数和一个偶数相加是合数，易知4和5
所以n<5时，不满足，输出-1
n>=5时，中间为4和5，左半部分为剩下的偶数，右半部分为剩下的奇数
代码如下：
```

```cpp
#include<bits/stdc++.h>
using namespace std;
#define ll long long


void solve(){
    ll n;
    cin>>n;
    if(n<=4){
        cout<<-1<<endl;
        return;
    }
    for(int i=2;i<=n;i+=2){
        if(i==4)
            continue;
        cout<<i<<" ";
    }
    cout<<4<<" "<<5<<" ";
    for(int i=1;i<=n;i+=2){
        if(i==5)
            continue;
        cout<<i<<" ";
    }
    cout<<endl;
    return;
}

int main(){
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    int tt=1;
    cin>>tt;
    while(tt--){
        solve();
    }
    return 0;
}
```
